\begin{definition}
\label{def:base}
\lean{Leantest.Basic.isTopoBase}
\leanok
Sea $\mathcal{T}$ una topología. Una \emph{base de la topología} $\mathcal{T}$ es una colección de abiertos $\mathcal{B} \subset \mathcal{T}$ de forma que cada abierto de $\mathcal{T}$ es unión de abiertos de $\mathcal{B}.
\end{definition}
